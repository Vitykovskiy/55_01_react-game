**Архитектура FSD**
   src/
   ├── app/                    # Инициализация приложения
   │   ├── routes/            # Маршрутизация
   │   ├── styles/            # Глобальные стили
   │   ├── App.tsx            # Собирает ВСЕ слои вместе
   │   └── index.tsx          # Точка входа 
   │
   ├── pages/                 # Страницы
   │   ├── LoginPage          # Пример страницы
   │       ├── model          # Модель страницы
   │       ├── ui             # Компоненты страницы
   │       ├── api            # API слой страницы
   │       └── index.tsx      # Public API
   │   └── index.tsx          # Реэкспорт
   │
   ├── shared/                # Общие модули
   │   ├── ui/                # Кнопки, инпуты - НЕ знают о бизнесе
   │   ├── lib/               # HTTP-клиент, утилиты
   │   └── config/            # Настройки
